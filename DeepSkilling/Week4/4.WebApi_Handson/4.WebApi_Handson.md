# 4.WebApi_Handson - Web API CRUD Operations & FromBody Binding

## Objectives & Conceptual Overview

### 1. Web API Data Operations (CRUD)
- **HTTP Verbs Mapping:**
  - **`GET`:** Retrieves existing resource data (all records or single record by ID).
  - **`POST`:** Creates a new resource record.
  - **`PUT`:** Updates an existing resource record based on user input.
  - **`DELETE`:** Removes a resource record from the data store.
- **Model Payload Extraction (`[FromBody]`):**
  The `[FromBody]` attribute instructs the model binding framework to extract JSON request body stream data and deserialize it into a target C# class (`Employee`).

---

### 2. HTTP PUT Action Method Validation Rules
- **Rule 1: ID Validation (`id <= 0`)**
  If the `id` value received in the request is less than or equal to `0`, the API immediately returns `BadRequest("Invalid employee id")` (HTTP 400).
- **Rule 2: Existence Validation (`id > 0` but Not Found)**
  If the `id` is positive but not present in the hardcoded list of employee records, the API returns `BadRequest("Invalid employee id")` (HTTP 400).
- **Rule 3: Record Update & Output Filtering**
  If the `id` is valid and present in the list, the action method updates the target record properties using the JSON body data (`emp`), filters the hardcoded list for the target `id`, and returns the updated `Employee` object wrapped in `ActionResult<Employee>` (HTTP 200 OK).

### 4. `Controllers/EmployeeController.cs`
```csharp
using Microsoft.AspNetCore.Http;
using Microsoft.AspNetCore.Mvc;
using System;
using System.Collections.Generic;
using System.Linq;
using WebApiHandsOn4.Models;

namespace WebApiHandsOn4.Controllers
{
    [Route("api/[controller]")]
    [ApiController]
    public class EmployeeController : ControllerBase
    {
        private static List<Employee> _employees = new List<Employee>();

        public EmployeeController()
        {
            if (!_employees.Any())
            {
                _employees = GetStandardEmployeeList();
            }
        }

        private List<Employee> GetStandardEmployeeList()
        {
            return new List<Employee>
            {
                new Employee
                {
                    Id = 1,
                    Name = "John Doe",
                    Salary = 75000,
                    Permanent = true,
                    Department = new Department { Id = 101, Name = "IT Infrastructure" },
                    Skills = new List<Skill>
                    {
                        new Skill { Id = 1, Name = "C#" },
                        new Skill { Id = 2, Name = "ASP.NET Core" },
                        new Skill { Id = 3, Name = "SQL Server" }
                    },
                    DateOfBirth = new DateTime(1995, 5, 15)
                },
                new Employee
                {
                    Id = 2,
                    Name = "Jane Smith",
                    Salary = 85000,
                    Permanent = true,
                    Department = new Department { Id = 102, Name = "Software Development" },
                    Skills = new List<Skill>
                    {
                        new Skill { Id = 4, Name = "Angular" },
                        new Skill { Id = 5, Name = "TypeScript" },
                        new Skill { Id = 6, Name = "REST Web API" }
                    },
                    DateOfBirth = new DateTime(1992, 8, 22)
                },
                new Employee
                {
                    Id = 3,
                    Name = "Robert Johnson",
                    Salary = 62000,
                    Permanent = false,
                    Department = new Department { Id = 103, Name = "Human Resources" },
                    Skills = new List<Skill>
                    {
                        new Skill { Id = 7, Name = "Recruitment" },
                        new Skill { Id = 8, Name = "Payroll" }
                    },
                    DateOfBirth = new DateTime(1998, 11, 3)
                }
            };
        }

        [HttpGet]
        [ProducesResponseType(StatusCodes.Status200OK, Type = typeof(List<Employee>))]
        public ActionResult<List<Employee>> Get()
        {
            return Ok(_employees);
        }

        [HttpGet("{id:int}")]
        [ProducesResponseType(StatusCodes.Status200OK, Type = typeof(Employee))]
        [ProducesResponseType(StatusCodes.Status400BadRequest)]
        public ActionResult<Employee> GetById(int id)
        {
            if (id <= 0) return BadRequest("Invalid employee id");
            var emp = _employees.FirstOrDefault(e => e.Id == id);
            if (emp == null) return BadRequest("Invalid employee id");
            return Ok(emp);
        }

        [HttpPost]
        [ProducesResponseType(StatusCodes.Status201Created, Type = typeof(Employee))]
        [ProducesResponseType(StatusCodes.Status400BadRequest)]
        public ActionResult<Employee> CreateEmployee([FromBody] Employee emp)
        {
            if (emp == null) return BadRequest("Invalid employee data");
            if (emp.Id <= 0) emp.Id = _employees.Any() ? _employees.Max(e => e.Id) + 1 : 1;
            _employees.Add(emp);
            return CreatedAtAction(nameof(GetById), new { id = emp.Id }, emp);
        }

        [HttpPut]
        [ProducesResponseType(StatusCodes.Status200OK, Type = typeof(Employee))]
        [ProducesResponseType(StatusCodes.Status400BadRequest)]
        public ActionResult<Employee> UpdateEmployee([FromBody] Employee emp)
        {
            if (emp == null || emp.Id <= 0)
            {
                return BadRequest("Invalid employee id");
            }

            var existingEmployee = _employees.FirstOrDefault(e => e.Id == emp.Id);
            if (existingEmployee == null)
            {
                return BadRequest("Invalid employee id");
            }

            existingEmployee.Name = emp.Name;
            existingEmployee.Salary = emp.Salary;
            existingEmployee.Permanent = emp.Permanent;
            existingEmployee.Department = emp.Department;
            existingEmployee.Skills = emp.Skills;
            existingEmployee.DateOfBirth = emp.DateOfBirth;

            var updatedEmployee = _employees.First(e => e.Id == emp.Id);
            return Ok(updatedEmployee);
        }

        [HttpPut("{id:int}")]
        [ProducesResponseType(StatusCodes.Status200OK, Type = typeof(Employee))]
        [ProducesResponseType(StatusCodes.Status400BadRequest)]
        public ActionResult<Employee> UpdateEmployeeWithId(int id, [FromBody] Employee emp)
        {
            if (id <= 0)
            {
                return BadRequest("Invalid employee id");
            }

            var existingEmployee = _employees.FirstOrDefault(e => e.Id == id);
            if (existingEmployee == null)
            {
                return BadRequest("Invalid employee id");
            }

            if (emp != null)
            {
                existingEmployee.Name = emp.Name;
                existingEmployee.Salary = emp.Salary;
                existingEmployee.Permanent = emp.Permanent;
                existingEmployee.Department = emp.Department;
                existingEmployee.Skills = emp.Skills;
                existingEmployee.DateOfBirth = emp.DateOfBirth;
            }

            var updatedEmployee = _employees.First(e => e.Id == id);
            return Ok(updatedEmployee);
        }

        [HttpDelete("{id:int}")]
        [ProducesResponseType(StatusCodes.Status200OK)]
        [ProducesResponseType(StatusCodes.Status400BadRequest)]
        public ActionResult DeleteEmployee(int id)
        {
            if (id <= 0)
            {
                return BadRequest("Invalid employee id");
            }

            var emp = _employees.FirstOrDefault(e => e.Id == id);
            if (emp == null)
            {
                return BadRequest("Invalid employee id");
            }

            _employees.Remove(emp);
            return Ok(new { message = $"Employee with Id = {id} successfully deleted." });
        }
    }
}
```

### 5. `Program.cs`
```csharp
using Microsoft.AspNetCore.Builder;
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Hosting;
using Microsoft.OpenApi.Models;

var builder = WebApplication.CreateBuilder(args);

builder.Services.AddControllers();
builder.Services.AddEndpointsApiExplorer();
builder.Services.AddSwaggerGen(c =>
{
    c.SwaggerDoc("v1", new OpenApiInfo
    {
        Title = "4.WebApi_Handson API",
        Version = "v1",
        Description = "ASP.NET Core 8.0 Web API hands-on demonstrating Employee CRUD operations (GET, POST, PUT, DELETE), [FromBody] model binding, and custom ID validation."
    });
});

var app = builder.Build();

app.UseSwagger();
app.UseSwaggerUI(c =>
{
    c.SwaggerEndpoint("/swagger/v1/swagger.json", "4.WebApi_Handson API v1");
    c.RoutePrefix = "swagger";
});

app.UseAuthorization();
app.MapControllers();
app.Run();
```
